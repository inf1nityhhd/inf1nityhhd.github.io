class Solution {
    public int numSubarraysWithSum(int[] A, int S) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);
        int curSum = 0;
        int count = 0;
        for(int num : A){
            curSum += num;
            if(map.containsKey(curSum - S)){
                count += map.get(curSum - S);
            }
            map.put(curSum, map.getOrDefault(curSum, 0) + 1);
        }
        return count;
    }
}
