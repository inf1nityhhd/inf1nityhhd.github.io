class Solution {
    public int minimumEffortPath(int[][] heights) {
        int row = heights.length;
        int col = heights[0].length;
        int[][] table = new int[row][col];
        int[] dx = new int[]{1, 0, -1, 0};
        int[] dy = new int[]{0, 1, 0, -1};
        for(int[] arr : table){
            Arrays.fill(arr, Integer.MAX_VALUE);
        }

        table[0][0] = 0;
        Queue<int[]> queue = new LinkedList<>();
        queue.add(new int[]{0, 0});

        while(!queue.isEmpty()){
            int[] node = queue.poll();
            for(int i = 0; i < 4; i++){
                int nx = node[0] + dx[i];
                int ny = node[1] + dy[i];
                if(nx < 0 || ny < 0 || nx >= row || ny >= col){
                    continue;
                }
                int cur_dist = Math.max(Math.abs(heights[nx][ny] - heights[node[0]][node[1]]), table[node[0]][node[1]]);
                //å‰ªææ¡ä»¶ï¼Œå¦‚æœå½“å‰è·¯å¾„åˆ°è¾¾ä¸‹ä¸€ç‚¹çš„æ¶ˆè€—å°äºä»¥å¾€è·¯å¾„çš„æ¶ˆè€—ï¼Œé‚£ä¹ˆå¯ä»¥ç»§ç»­éå†ã€‚
                if(cur_dist < table[nx][ny]){
                    table[nx][ny] = cur_dist;
                    queue.add(new int[]{nx, ny});
                }
            }
        }
        return table[row - 1][col - 1];
    }
}
