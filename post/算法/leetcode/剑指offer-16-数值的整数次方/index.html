class Solution {
    public double myPow(double x, int n) {
        if(x == 0){
            return 0;
        }
        long newN = n;
        double res = 1.0;
        if(newN < 0) {
            x = 1 / x;
            newN = -newN;
        }
        while(newN > 0) {
            if((newN & 1) == 1){
                res *= x;
            }
            x *= x;
            newN /= 2;
        }
        return res;
    }
}
